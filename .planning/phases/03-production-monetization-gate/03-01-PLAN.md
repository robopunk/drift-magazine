    Create `.github/workflows/agent-run.yml` with the following exact content (per D-06, D-07, D-08):

    ```yaml
    name: Drift Agent — Bi-weekly Research Run

    on:
      schedule:
        # Bi-weekly: 1st and 15th of each month at 06:00 UTC
        - cron: "0 6 1,15 * *"
      workflow_dispatch:
        inputs:
          company_id:
            description: "Optional: Run for a specific company ID"
            required: false
            type: string

    jobs:
      research-run:
        runs-on: ubuntu-latest
        timeout-minutes: 30

        steps:
          - name: Checkout repository
            uses: actions/checkout@v4

          - name: Set up Python 3.11
            uses: actions/setup-python@v5
            with:
              python-version: "3.11"
              cache: "pip"
              cache-dependency-path: backend/requirements.txt

          - name: Install dependencies
            run: pip install -r backend/requirements.txt

          - name: Run research agent
            env:
              ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
              SUPABASE_URL: ${{ secrets.SUPABASE_URL }}
              SUPABASE_SERVICE_KEY: ${{ secrets.SUPABASE_SERVICE_KEY }}
              FIRECRAWL_API_KEY: ${{ secrets.FIRECRAWL_API_KEY }}
            run: |
              if [ -n "${{ github.event.inputs.company_id }}" ]; then
                python backend/agent.py --company-id "${{ github.event.inputs.company_id }}"
              else
                python backend/agent.py
              fi
    ```

    Key design decisions:
    - Cron runs on 1st and 15th at 06:00 UTC (bi-weekly per D-06)
    - `workflow_dispatch` allows manual trigger with optional company_id (per D-08)
    - GitHub's built-in email notification on failure serves as alerting (per D-07)
    - Python 3.11 with pip caching for faster runs
    - 30-minute timeout prevents runaway agent executions
    - All four secrets referenced: ANTHROPIC_API_KEY, SUPABASE_URL, SUPABASE_SERVICE_KEY, FIRECRAWL_API_KEY
